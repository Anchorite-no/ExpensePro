const CustomOrganicNetwork = ({ transactions }) => {
  const [nodes, setNodes] = useState([]);
  const [links, setLinks] = useState([]);

  useEffect(() => {
    const nodesMap = {}; const linksMap = {};
    transactions.forEach(t => {
      t.tags.forEach(tag => { if (!nodesMap[tag]) nodesMap[tag] = { id: tag, count: 0 }; nodesMap[tag].count += 1; });
      for (let i = 0; i < t.tags.length; i++) {
        for (let j = i + 1; j < t.tags.length; j++) {
          const pair = [t.tags[i], t.tags[j]].sort().join('||');
          linksMap[pair] = (linksMap[pair] || 0) + 1;
        }
      }
    });

    const initNodes = Object.values(nodesMap).sort((a, b) => b.count - a.count).slice(0, 20); 
    const validNodeIds = new Set(initNodes.map(n => n.id));
    const initLinks = Object.entries(linksMap).map(([pair, weight]) => {
      const [source, target] = pair.split('||'); return { source, target, weight };
    }).filter(l => validNodeIds.has(l.source) && validNodeIds.has(l.target));

    const centerX = 400, centerY = 160;
    initNodes.forEach((node, i) => {
      node.x = centerX + (Math.random() - 0.5) * 400; 
      node.y = centerY + (Math.random() - 0.5) * 150;
      node.vx = 0; node.vy = 0;
      node.color = COLORS[i % COLORS.length];
      node.r = Math.max(14, Math.min(34, 10 + node.count * 3));
      node.animDelay = Math.random() * 2;
      node.animDuration = 3 + Math.random() * 2;
    });

    for (let iter = 0; iter < 120; iter++) {
      for (let i = 0; i < initNodes.length; i++) {
        for (let j = i + 1; j < initNodes.length; j++) {
          let dx = initNodes[i].x - initNodes[j].x, dy = initNodes[i].y - initNodes[j].y;
          let dist = Math.sqrt(dx*dx + dy*dy) || 1;
          let force = 6000 / dist; 
          initNodes[i].vx += (dx/dist)*force; initNodes[i].vy += (dy/dist)*force;
          initNodes[j].vx -= (dx/dist)*force; initNodes[j].vy -= (dy/dist)*force;
        }
      }
      initLinks.forEach(link => {
        let source = initNodes.find(n => n.id === link.source), target = initNodes.find(n => n.id === link.target);
        if(source && target) {
          let dx = source.x - target.x, dy = source.y - target.y;
          let dist = Math.sqrt(dx*dx + dy*dy) || 1;
          let force = (dist*dist) / 120 * (link.weight * 0.3); 
          source.vx -= (dx/dist)*force; source.vy -= (dy/dist)*force;
          target.vx += (dx/dist)*force; target.vy += (dy/dist)*force;
        }
      });
      initNodes.forEach(n => {
        n.vx += (centerX - n.x) * (0.01 + n.count * 0.004); 
        n.vy += (centerY - n.y) * (0.02 + n.count * 0.008);
        n.x += n.vx * 0.05; n.y += n.vy * 0.05;
        n.vx = 0; n.vy = 0;
      });
    }
    setNodes([...initNodes]); setLinks(initLinks);
  }, [transactions]);

  return (
    <div className="w-full overflow-x-auto flex justify-center custom-scrollbar">
      {/* é€‚é…æµ…è‰²ä¸»é¢˜çš„æ˜Ÿç³»ç½‘ç»œ */}
      <svg width="100%" height="320" viewBox="0 0 800 320" className="font-sans min-w-[800px]">
        <defs>
          <filter id="glow" x="-20%" y="-20%" width="140%" height="140%">
            <feGaussianBlur stdDeviation="3" result="blur" />
            <feMerge><feMergeNode in="blur"/><feMergeNode in="SourceGraphic"/></feMerge>
          </filter>
        </defs>

        {/* æµ…è‰²è¾…åŠ©è½¨é“ */}
        <ellipse cx="400" cy="160" rx="140" ry="70" fill="none" stroke="#e2e8f0" strokeDasharray="4 4" strokeOpacity="0.8" />
        <ellipse cx="400" cy="160" rx="260" ry="120" fill="none" stroke="#e2e8f0" strokeDasharray="4 4" strokeOpacity="0.6" />
        <ellipse cx="400" cy="160" rx="380" ry="160" fill="none" stroke="#e2e8f0" strokeDasharray="4 4" strokeOpacity="0.3" />

        {links.map((link, i) => {
          const source = nodes.find(n => n.id === link.source), target = nodes.find(n => n.id === link.target);
          if (!source || !target) return null;
          const dx = target.x - source.x, dy = target.y - source.y;
          const cx = source.x + dx / 2 - dy * 0.2; 
          const cy = source.y + dy / 2 + dx * 0.2;
          return (
            <path key={i} d={`M ${source.x} ${source.y} Q ${cx} ${cy} ${target.x} ${target.y}`} fill="none" stroke="#94a3b8" strokeWidth={link.weight * 1.5} strokeOpacity={0.4} />
          )
        })}

        {nodes.map((node) => (
          <g key={node.id} className="cursor-pointer">
            <animateTransform attributeName="transform" type="translate" values={`${node.x},${node.y}; ${node.x},${node.y - 6}; ${node.x},${node.y}`} dur={`${node.animDuration}s`} begin={`${node.animDelay}s`} repeatCount="indefinite" />
            <circle r={node.r} fill={node.color} fillOpacity="0.95" stroke="#fff" strokeWidth="2.5" filter="url(#glow)" />
            {/* æµ…è‰²ä¸»é¢˜ä¸‹æ–‡å­—ä¿æŒç™½è‰²ä»¥ä¾¿åœ¨äº®è‰²æ°”æ³¡ä¸Šå¯è¯»ï¼Œæˆ–åŠ è½»å¾®é˜´å½± */}
            <text x="0" y="3" textAnchor="middle" fill="#fff" fontSize={node.r > 18 ? 12 : 10} fontWeight="bold" className="select-none" style={{textShadow: '0px 1px 2px rgba(0,0,0,0.2)'}}>{node.id}</text>
          </g>
        ))}
      </svg>
    </div>
  );
};
